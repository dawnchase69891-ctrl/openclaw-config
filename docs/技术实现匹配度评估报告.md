#### 4.1.2 任务队列管理器

**建议实现**:
```python
# scripts/task_queue_manager.py
import asyncio
from typing import Dict, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta

@dataclass
class Task:
    agent_id: str
    task: str
    priority: int
    created_at: datetime
    timeout: int = 300  # 5 分钟超时

class TaskQueueManager:
    def __init__(self, max_concurrent: int = 5):
        self.max_concurrent = max_concurrent
        self.running = 0
        self.queue: asyncio.PriorityQueue = asyncio.PriorityQueue()
        self.lock = asyncio.Lock()
        self.active_tasks: Dict[str, Task] = {}
    
    async def enqueue(self, task: Task):
        """添加任务到队列"""
        await self.queue.put((-task.priority, task))  # 负数实现最大堆
    
    async def spawn_with_control(self, agent_id: str, task: str, priority: int = 1):
        """带并发控制的 spawn"""
        t = Task(agent_id, task, priority, datetime.now())
        
        async with self.lock:
            if self.running >= self.max_concurrent:
                # 队列等待
                await self.queue.put((-priority, t))
                return
            
            self.running += 1
            self.active_tasks[agent_id] = t
        
        try:
            # 设置超时
            result = await asyncio.wait_for(
                sessions_spawn(agent_id=agent_id, task=task),
                timeout=t.timeout
            )
            return result
        except asyncio.TimeoutError:
            # 超时处理
            await self.handle_timeout(t)
            raise
        finally:
            async with self.lock:
                self.running -= 1
                del self.active_tasks[agent_id]
                # 检查等待队列
                if not self.queue.empty():
                    _, next_task = await self.queue.get()
                    asyncio.create_task(
                        self.spawn_with_control(next_task.agent_id, next_task.task, -next_task.priority)
                    )
```
